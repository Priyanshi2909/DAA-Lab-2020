#include<stdio.h>
#include<conio.h>

int max(int a, int b)
{
	return (a > b) ? a : b;
}
int knapSack(int W, int wt[], int val[], int n)
{
	if (n == 0 || W == 0)
		return 0;
	if (wt[n - 1] > W)
		return knapSack(W, wt, val, n - 1);
	else
		return max(
			val[n - 1] + knapSack(W - wt[n - 1], wt, val, n - 1),
			knapSack(W, wt, val, n - 1));
}
void main()
{
	int val[15],wt[15],i,W,n;
	printf("Enter the number of elements: ");
	scanf("%d",&n);
	printf("Enter the value:\n");
	for(i=0;i<n;i++){
		scanf("%d",&val[i]);
	}
	printf("Enter the weight:\n");
	for(i=0;i<n;i++){
		scanf("%d",&wt[i]);
	}
	printf("Enter the weight of bag: ");
	scanf("%d",&W);
	printf("\n Total Profit: %d", knapSack(W, wt, val, n));
	getch();
}
